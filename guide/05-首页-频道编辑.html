              3、修改添加频道的处理逻辑



               
               
               
               
               
               
               
               
               
               
               
               
               
               
               


              async onAddChannel (channel) {
    this.myChannels.push(channel)
    // 数据持久化处理
    if (this.user) {
        try {
          // 已登录，把数据请求接口放到线上
          await addUserChannel({
            id: channel.id, // 频道ID
            seq: this.myChannels.length // 序号
          })
        } catch (err) {
          this.$toast('保存失败，请稍后重试')
        }
    } else {
        // 未登录，把数据存储到本地
        setItem('TOUTIAO_CHANNELS', this.myChannels)
    }
}
